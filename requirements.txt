# requirements.txt — Spaceflight–Disease Network Separation Pipeline
#
# Install with:
#   pip install -r requirements.txt
#
# For a fully reproducible conda environment, use environment.yml instead.
#
# ── Python version ────────────────────────────────────────────────────────────
# Python 3.10 or 3.11 is strongly recommended.
#
# NetMedPy's own package metadata declares incompatibility with Python 3.12
# due to its Ray dependency (see github.com/menicgiulia/NetMedPy). Although
# Ray itself added Python 3.12 support from v2.31.0 onward, NetMedPy has not
# updated its Requires-Python ceiling as of this writing. Use Python <=3.11
# until a NetMedPy release explicitly lifts the restriction.

# ── Core scientific stack ─────────────────────────────────────────────────────
numpy>=1.24,<3.0
pandas>=2.0,<3.0
scipy>=1.11,<2.0
statsmodels>=0.14,<1.0

# ── Network analysis ──────────────────────────────────────────────────────────
networkx>=3.1,<4.0

# ── Network medicine ──────────────────────────────────────────────────────────
# Pin to 0.1.x: the separation_z_score API may change in 0.2.x.
# NetMedPy currently declares Requires-Python >=3.8,<3.12 — see note above.
netmedpy>=0.1.4,<0.2.0

# ── Ontology parsing ──────────────────────────────────────────────────────────
# pronto 2.x completely rewrote the API vs 1.x (different class hierarchy,
# .obsolete attribute renamed). Latest stable: 2.7.3.
pronto>=2.5,<3.0

# ── Parallelism ───────────────────────────────────────────────────────────────
# Ray 2.x changed the init() signature vs 1.x — do not install 1.x.
# NetMedPy 0.1.x calls ray._private.utils.get_ray_temp_dir,
# removed in Ray 2.20. Pin to last compatible release.
ray==2.10.0

# ── Gene ID mapping ───────────────────────────────────────────────────────────
# Optional fallback; primary mapping uses the local NCBI gene_info file.
mygene>=3.2,<4.0

# ── File I/O ──────────────────────────────────────────────────────────────────
openpyxl>=3.1,<4.0     # .xlsx read/write (pandas read_excel backend)
joblib>=1.3,<2.0       # legacy pickle fallback in utils.load()

# ── System utilities ──────────────────────────────────────────────────────────
psutil>=5.9,<7.0

# ── Visualisation ─────────────────────────────────────────────────────────────
matplotlib>=3.7,<4.0
